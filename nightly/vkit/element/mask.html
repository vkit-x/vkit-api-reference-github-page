502    def to_disconnected_polygons(
503        self,
504        cv_find_contours_method: int = cv.CHAIN_APPROX_SIMPLE,
505    ) -> Sequence['Polygon']:
506        # [ (N, 1, 2), ... ]
507        cv_contours, _ = cv.findContours(
508            (self.np_mask.astype(np.uint8) * 255),
509            cv.RETR_LIST,
510            cv_find_contours_method,
511        )
512
513        polygons: List[Polygon] = []
514
515        # Ignore logging of shapely.geos.
516        shapely_geos_logger = logging.getLogger('shapely.geos')
517        shapely_geos_logger_level = shapely_geos_logger.level
518        shapely_geos_logger.setLevel(logging.WARNING)
519
520        for cv_contour in cv_contours:
521            assert cv_contour.shape[1] == 1
522            np_points = np.squeeze(cv_contour, axis=1)
523
524            if self.box:
525                np_points[:, 0] += self.box.left
526                np_points[:, 1] += self.box.up
527
528            if np_points.shape[0] < 3:
529                # If less than 3 points, ignore.
530                continue
531
532            polygon = Polygon.from_np_array(np_points)
533
534            # Split further based on shapley library,
535            # since some contours generated by opencv is consider invalid in shapely.
536            shapely_polygon = polygon.to_shapely_polygon()
537            shapely_valid_geom = shapely_make_valid(shapely_polygon)
538
539            if isinstance(shapely_valid_geom, ShapelyPolygon):
540                polygons.append(polygon)
541
542            elif isinstance(shapely_valid_geom, (ShapelyMultiPolygon, ShapelyGeometryCollection)):
543                for shapely_geom in shapely_valid_geom.geoms:
544                    if isinstance(shapely_geom, ShapelyPolygon):
545                        polygons.append(Polygon.from_shapely_polygon(shapely_geom))
546                    elif isinstance(shapely_geom, ShapelyMultiPolygon):
547                        # I don't know why, but this do happen.
548                        for shapely_sub_geom in shapely_geom.geoms:
549                            if isinstance(shapely_sub_geom, ShapelyPolygon):
550                                polygons.append(Polygon.from_shapely_polygon(shapely_sub_geom))
551                            else:
552                                logger.debug(f'ignore shapely_sub_geom={shapely_sub_geom}')
553                    else:
554                        logger.debug(f'ignore shapely_geom={shapely_geom}')
555
556            else:
557                logger.debug(f'ignore shapely_valid_geom={shapely_valid_geom}')
558
559        # Reset logging level.
560        shapely_geos_logger.setLevel(shapely_geos_logger_level)
561
562        return polygons

581def generate_fill_by_masks_mask(
582    shape: Tuple[int, int],
583    masks: Iterable[Mask],
584    mode: FillByElementsMode,
585):
586    if mode == FillByElementsMode.UNION:
587        return None
588
589    masks_mask = Mask.from_shape(shape)
590
591    with masks_mask.writable_context:
592        for mask in masks:
593            if mask.box:
594                boxed_mat = mask.box.extract_np_array(masks_mask.mat)
595            else:
596                boxed_mat = masks_mask.mat
597
598            np_non_oob_mask = (boxed_mat < 255)
599            boxed_mat[mask.np_mask & np_non_oob_mask] += 1
600
601        if mode == FillByElementsMode.DISTINCT:
602            masks_mask.mat[masks_mask.mat > 1] = 0
603
604        elif mode == FillByElementsMode.INTERSECT:
605            masks_mask.mat[masks_mask.mat == 1] = 0
606
607        else:
608            raise NotImplementedError()
609
610    return masks_mask
