 186def calculate_boxed_masks_intersected_ratio(
 187    anchor_mask: Mask,
 188    candidate_mask: Mask,
 189    use_candidate_as_base: bool = False,
 190):
 191    anchor_box = anchor_mask.box
 192    assert anchor_box
 193
 194    candidate_box = candidate_mask.box
 195    assert candidate_box
 196
 197    # Calculate intersection.
 198    up = max(anchor_box.up, candidate_box.up)
 199    down = min(anchor_box.down, candidate_box.down)
 200    left = max(anchor_box.left, candidate_box.left)
 201    right = min(anchor_box.right, candidate_box.right)
 202
 203    if up > down or left > right:
 204        return 0.0
 205
 206    np_intersected_anchor_mask = anchor_mask.mat[
 207        up - anchor_box.up:down - anchor_box.up + 1,
 208        left - anchor_box.left:right - anchor_box.left + 1,
 209    ]  # yapf: disable
 210    np_intersected_candidate_mask = candidate_mask.mat[
 211        up - candidate_box.up:down - candidate_box.up + 1,
 212        left - candidate_box.left:right - candidate_box.left + 1,
 213    ]  # yapf: disable
 214    np_intersected_mask = np_intersected_anchor_mask & np_intersected_candidate_mask
 215    intersected_area = int(np_intersected_mask.sum())
 216
 217    if use_candidate_as_base:
 218        base_area = int(candidate_mask.np_mask.sum())
 219    else:
 220        base_area = (
 221            int(anchor_mask.np_mask.sum()) + int(candidate_mask.np_mask.sum()) - intersected_area
 222        )
 223
 224    return intersected_area / base_area

 229    @classmethod
 230    def patch_text_region_polygons(
 231        cls,
 232        text_region_polygons: Sequence[Polygon],
 233        grouped_char_polygons: Optional[Sequence[Sequence[Polygon]]],
 234    ):
 235        if grouped_char_polygons is None:
 236            return text_region_polygons
 237
 238        assert len(text_region_polygons) == len(grouped_char_polygons)
 239
 240        patched_text_region_polygons: List[Polygon] = []
 241        for text_region_polygon, char_polygons in zip(text_region_polygons, grouped_char_polygons):
 242            # Need to make sure all char polygons are included.
 243            unionized_polygons = [text_region_polygon]
 244            unionized_polygons.extend(char_polygons)
 245
 246            bounding_box = Box.from_boxes((polygon.bounding_box for polygon in unionized_polygons))
 247            mask = Mask.from_shapable(bounding_box).to_box_attached(bounding_box)
 248            for polygon in unionized_polygons:
 249                polygon.fill_mask(mask)
 250
 251            patched_text_region_polygons.append(mask.to_external_polygon())
 252
 253        return patched_text_region_polygons

 255    @classmethod
 256    def process_text_region_polygons(
 257        cls,
 258        text_region_polygon_dilate_ratio: float,
 259        shape: Tuple[int, int],
 260        text_region_polygons: Sequence[Polygon],
 261        force_no_dilation_flags: Optional[Sequence[bool]] = None,
 262    ):
 263        text_mask = Mask.from_polygons(shape, text_region_polygons)
 264        non_text_mask = text_mask.to_inverted_mask()
 265
 266        box = Box.from_shape(shape)
 267        text_mask = text_mask.to_box_attached(box)
 268        non_text_mask = non_text_mask.to_box_attached(box)
 269
 270        bounding_extended_text_region_masks: List[Mask] = []
 271        bounding_rectangular_polygons: List[Polygon] = []
 272
 273        if force_no_dilation_flags is None:
 274            force_no_dilation_flags_iter = itertools.repeat(False)
 275        else:
 276            assert len(force_no_dilation_flags) == len(text_region_polygons)
 277            force_no_dilation_flags_iter = force_no_dilation_flags
 278
 279        for text_region_polygon, force_no_dilation_flag in zip(
 280            text_region_polygons, force_no_dilation_flags_iter
 281        ):
 282            original_text_region_polygon = text_region_polygon
 283
 284            if not force_no_dilation_flag:
 285                # Dilate.
 286                text_region_polygon = text_region_polygon.to_dilated_polygon(
 287                    ratio=text_region_polygon_dilate_ratio,
 288                )
 289                text_region_polygon = text_region_polygon.to_clipped_polygon(shape)
 290
 291            # Get bounding rectangular box (polygon).
 292            bounding_rectangular_polygon = \
 293                text_region_polygon.to_bounding_rectangular_polygon(shape)
 294
 295            bounding_box = bounding_rectangular_polygon.bounding_box
 296
 297            # Get other text region.
 298            bounding_other_text_mask = bounding_rectangular_polygon.extract_mask(text_mask).copy()
 299            # NOTE: Use the original text region polygon to unset the current text mask.
 300            original_text_region_polygon.fill_mask(bounding_other_text_mask, 0)
 301
 302            # Get protentially dilated text region.
 303            bounding_text_mask = Mask.from_shapable(bounding_other_text_mask)
 304            bounding_text_mask = bounding_text_mask.to_box_attached(bounding_box)
 305            # NOTE: Use the protentially dilated text region polygon to set the current text mask.
 306            text_region_polygon.fill_mask(bounding_text_mask, value=1)
 307
 308            # Should not use the protentially dilated text region polygon anymore.
 309            del text_region_polygon
 310
 311            # Trim protentially dilated text region polygon by eliminating other text region.
 312            bounding_trimmed_text_mask = Mask.from_masks(
 313                bounding_box,
 314                [
 315                    # Includes the protentially dilated text region.
 316                    bounding_text_mask,
 317                    # But not includes any other text regions.
 318                    bounding_other_text_mask.to_inverted_mask(),
 319                ],
 320                ElementSetOperationMode.INTERSECT,
 321            )
 322
 323            # Get non-text region.
 324            bounding_non_text_mask = bounding_rectangular_polygon.extract_mask(non_text_mask)
 325
 326            # Combine trimmed text region and non-text region.
 327            bounding_extended_text_region_mask = Mask.from_masks(
 328                bounding_box,
 329                [bounding_trimmed_text_mask, bounding_non_text_mask],
 330            )
 331
 332            bounding_extended_text_region_masks.append(bounding_extended_text_region_mask)
 333            bounding_rectangular_polygons.append(bounding_rectangular_polygon)
 334
 335        return bounding_extended_text_region_masks, bounding_rectangular_polygons

 337    @classmethod
 338    def analyze_bounding_rectangular_polygons(
 339        cls,
 340        bounding_rectangular_polygons: Sequence[Polygon],
 341    ):
 342        long_side_ratios: List[float] = []
 343        long_side_angles: List[int] = []
 344
 345        for polygon in bounding_rectangular_polygons:
 346            # Get reference line.
 347            point0, point1, _, point3 = polygon.points
 348            side0_length = math.hypot(
 349                point0.smooth_y - point1.smooth_y,
 350                point0.smooth_x - point1.smooth_x,
 351            )
 352            side1_length = math.hypot(
 353                point0.smooth_y - point3.smooth_y,
 354                point0.smooth_x - point3.smooth_x,
 355            )
 356
 357            long_side_ratios.append(
 358                max(side0_length, side1_length) / min(side0_length, side1_length)
 359            )
 360
 361            point_a = point0
 362            if side0_length > side1_length:
 363                # Reference line (p0 -> p1).
 364                point_b = point1
 365            else:
 366                # Reference line (p0 -> p3).
 367                point_b = point3
 368
 369            # Get angle of reference line in [0, 180)
 370            np_theta = np.arctan2(
 371                point_a.smooth_y - point_b.smooth_y,
 372                point_a.smooth_x - point_b.smooth_x,
 373            )
 374            np_theta = np_theta % np.pi
 375            angle = round(np_theta / np.pi * 180)
 376
 377            long_side_angles.append(angle)
 378
 379        return long_side_ratios, long_side_angles

 381    @classmethod
 382    def get_typical_angle(
 383        cls,
 384        typical_long_side_ratio_min: float,
 385        long_side_ratios: Sequence[float],
 386        long_side_angles: Sequence[int],
 387    ):
 388        typical_indices: Set[int] = set()
 389        typical_long_side_angles: List[float] = []
 390
 391        for idx, (long_side_ratio, long_side_angle) in \
 392                enumerate(zip(long_side_ratios, long_side_angles)):
 393            if long_side_ratio < typical_long_side_ratio_min:
 394                continue
 395
 396            typical_indices.add(idx)
 397            typical_long_side_angles.append(long_side_angle)
 398
 399        if not typical_long_side_angles:
 400            return None, typical_indices
 401
 402        np_angles = np.asarray(typical_long_side_angles) / 180 * np.pi
 403        np_sin_mean = np.sin(np_angles).mean()
 404        np_cos_mean = np.cos(np_angles).mean()
 405
 406        np_theta = np.arctan2(np_sin_mean, np_cos_mean)
 407        np_theta = np_theta % np.pi
 408        typical_angle = round(np_theta / np.pi * 180)
 409
 410        return typical_angle, typical_indices

 412    @classmethod
 413    def get_flattening_rotate_angles(
 414        cls,
 415        typical_angle: Optional[int],
 416        typical_indices: Set[int],
 417        long_side_angles: Sequence[int],
 418    ):
 419        if typical_angle is not None:
 420            assert typical_indices
 421
 422        flattening_rotate_angles: List[int] = []
 423
 424        for idx, long_side_angle in enumerate(long_side_angles):
 425            if typical_angle is None or idx in typical_indices:
 426                # Dominated by long_side_angle.
 427                main_angle = long_side_angle
 428
 429            else:
 430                # Dominated by typical_angle.
 431                short_side_angle = (long_side_angle + 90) % 180
 432                long_side_delta = abs((long_side_angle - typical_angle + 90) % 180 - 90)
 433                short_side_delta = abs((short_side_angle - typical_angle + 90) % 180 - 90)
 434
 435                if long_side_delta < short_side_delta:
 436                    main_angle = long_side_angle
 437                else:
 438                    main_angle = short_side_angle
 439
 440            # Angle for flattening.
 441            if main_angle <= 90:
 442                flattening_rotate_angle = 360 - main_angle
 443            else:
 444                flattening_rotate_angle = 180 - main_angle
 445            flattening_rotate_angles.append(flattening_rotate_angle)
 446
 447        return flattening_rotate_angles

 449    @classmethod
 450    def build_flattened_text_regions(
 451        cls,
 452        image: Image,
 453        text_region_polygons: Sequence[Polygon],
 454        bounding_extended_text_region_masks: Sequence[Mask],
 455        typical_indices: Set[int],
 456        flattening_rotate_angles: Sequence[int],
 457        grouped_char_polygons: Optional[Sequence[Sequence[Polygon]]],
 458    ):
 459        flattened_text_regions: List[FlattenedTextRegion] = []
 460
 461        for idx, (
 462            text_region_polygon,
 463            bounding_extended_text_region_mask,
 464            flattening_rotate_angle,
 465        ) in enumerate(
 466            zip(
 467                text_region_polygons,
 468                bounding_extended_text_region_masks,
 469                flattening_rotate_angles,
 470            )
 471        ):
 472            bounding_box = bounding_extended_text_region_mask.box
 473            assert bounding_box
 474
 475            # Extract image.
 476            text_region_image = bounding_extended_text_region_mask.extract_image(image)
 477
 478            # Shift char polygons.
 479            relative_char_polygons = None
 480            if grouped_char_polygons is not None:
 481                char_polygons = grouped_char_polygons[idx]
 482                relative_char_polygons = [
 483                    char_polygon.to_relative_polygon(
 484                        origin_y=bounding_box.up,
 485                        origin_x=bounding_box.left,
 486                    ) for char_polygon in char_polygons
 487                ]
 488
 489            # Rotate.
 490            rotated_result = rotate.distort(
 491                {'angle': flattening_rotate_angle},
 492                image=text_region_image,
 493                mask=bounding_extended_text_region_mask,
 494                polygons=relative_char_polygons,
 495            )
 496            rotated_text_region_image = rotated_result.image
 497            assert rotated_text_region_image
 498            rotated_bounding_extended_text_region_mask = rotated_result.mask
 499            assert rotated_bounding_extended_text_region_mask
 500            # Could be None.
 501            rotated_char_polygons = rotated_result.polygons
 502
 503            # Trim.
 504            rotated_trimmed_box = rotated_bounding_extended_text_region_mask.to_external_box()
 505
 506            trimmed_text_region_image = rotated_text_region_image.to_cropped_image(
 507                up=rotated_trimmed_box.up,
 508                down=rotated_trimmed_box.down,
 509                left=rotated_trimmed_box.left,
 510                right=rotated_trimmed_box.right,
 511            )
 512
 513            trimmed_mask = rotated_trimmed_box.extract_mask(
 514                rotated_bounding_extended_text_region_mask
 515            )
 516
 517            trimmed_char_polygons = None
 518            if rotated_char_polygons:
 519                trimmed_char_polygons = [
 520                    rotated_char_polygon.to_relative_polygon(
 521                        origin_y=rotated_trimmed_box.up,
 522                        origin_x=rotated_trimmed_box.left,
 523                    ) for rotated_char_polygon in rotated_char_polygons
 524                ]
 525
 526            flattened_text_regions.append(
 527                FlattenedTextRegion(
 528                    is_typical=(idx in typical_indices),
 529                    text_region_polygon=text_region_polygon,
 530                    text_region_image=bounding_extended_text_region_mask.extract_image(image),
 531                    bounding_extended_text_region_mask=bounding_extended_text_region_mask,
 532                    flattening_rotate_angle=flattening_rotate_angle,
 533                    rotated_trimmed_box=rotated_trimmed_box,
 534                    shape_before_resize=trimmed_text_region_image.shape,
 535                    post_rotate_angle=0,
 536                    flattened_image=trimmed_text_region_image,
 537                    flattened_mask=trimmed_mask,
 538                    flattened_char_polygons=trimmed_char_polygons,
 539                )
 540            )
 541
 542        return flattened_text_regions

 544    def __init__(
 545        self,
 546        typical_long_side_ratio_min: float,
 547        text_region_polygon_dilate_ratio: float,
 548        image: Image,
 549        text_region_polygons: Sequence[Polygon],
 550        grouped_char_polygons: Optional[Sequence[Sequence[Polygon]]] = None,
 551        is_training: bool = False,
 552    ):
 553        self.origional_text_region_polygons = text_region_polygons
 554
 555        self.text_region_polygons = self.patch_text_region_polygons(
 556            text_region_polygons=text_region_polygons,
 557            grouped_char_polygons=grouped_char_polygons,
 558        )
 559
 560        force_no_dilation_flags = None
 561        if is_training:
 562            assert grouped_char_polygons and len(text_region_polygons) == len(grouped_char_polygons)
 563            force_no_dilation_flags = []
 564            for char_polygons in grouped_char_polygons:
 565                force_no_dilation_flags.append(not char_polygons)
 566
 567        self.bounding_extended_text_region_masks, self.bounding_rectangular_polygons = \
 568            self.process_text_region_polygons(
 569                text_region_polygon_dilate_ratio=text_region_polygon_dilate_ratio,
 570                shape=image.shape,
 571                text_region_polygons=self.text_region_polygons,
 572                force_no_dilation_flags=force_no_dilation_flags,
 573            )
 574
 575        self.long_side_ratios, self.long_side_angles = \
 576            self.analyze_bounding_rectangular_polygons(self.bounding_rectangular_polygons)
 577

 600def build_background_image_for_stacking(height: int, width: int):
 601    np_rgb_rows = [np.zeros((width, 3), dtype=np.uint8) for _ in range(3)]
 602    rgb_tuples = [(255, 0, 0), (0, 255, 0), (0, 0, 255)]
 603
 604    for color_offset, np_row in enumerate(np_rgb_rows):
 605        for color_idx in range(3):
 606            color_tuple = rgb_tuples[(color_offset + color_idx) % 3]
 607            np_row[color_idx::3] = color_tuple
 608
 609    np_image = np.zeros((height, width, 3), dtype=np.uint8)
 610    for row_offset, np_row in enumerate(np_rgb_rows):
 611        np_image[row_offset::3] = np_row
 612
 613    return Image(mat=np_image)

 616def stack_flattened_text_regions(
 617    page_pad: int,
 618    flattened_text_regions_pad: int,
 619    flattened_text_regions: Sequence[FlattenedTextRegion],
 620):
 621    page_double_pad = 2 * page_pad
 622    flattened_text_regions_double_pad = 2 * flattened_text_regions_pad
 623
 624    rect_packer = RectPacker(rotation=False)
 625
 626    # Add box and bin.
 627    # NOTE: Only one bin is added, that is, packing all text region into one image.
 628    bin_width = 0
 629    bin_height = 0
 630
 631    for ftr_idx, flattened_text_region in enumerate(flattened_text_regions):
 632        rect_packer.add_rect(
 633            width=flattened_text_region.width + flattened_text_regions_double_pad,
 634            height=flattened_text_region.height + flattened_text_regions_double_pad,
 635            rid=ftr_idx,
 636        )
 637
 638        bin_width = max(bin_width, flattened_text_region.width)
 639        bin_height += flattened_text_region.height
 640
 641    bin_width += flattened_text_regions_double_pad
 642    bin_height += flattened_text_regions_double_pad
 643
 644    rect_packer.add_bin(width=bin_width, height=bin_height)
 645
 646    # Pack boxes.
 647    rect_packer.pack()  # type: ignore
 648
 649    # Get packed boxes.
 650    unordered_boxes: List[Box] = []
 651    ftr_indices: List[int] = []
 652    for bin_idx, x, y, width, height, ftr_idx in rect_packer.rect_list():
 653        assert bin_idx == 0
 654        unordered_boxes.append(Box(
 655            up=y,
 656            down=y + height - 1,
 657            left=x,
 658            right=x + width - 1,
 659        ))
 660        ftr_indices.append(ftr_idx)
 661
 662    # Order boxes.
 663    inverse_ftr_indices = [-1] * len(ftr_indices)
 664    for inverse_ftr_idx, ftr_idx in enumerate(ftr_indices):
 665        inverse_ftr_indices[ftr_idx] = inverse_ftr_idx
 666    for inverse_ftr_idx in inverse_ftr_indices:
 667        assert inverse_ftr_idx >= 0
 668    boxes = [unordered_boxes[inverse_ftr_idx] for inverse_ftr_idx in inverse_ftr_indices]
 669
 670    page_height = max(box.down for box in boxes) + 1 + page_double_pad
 671    page_width = max(box.right for box in boxes) + 1 + page_double_pad
 672
 673    image = build_background_image_for_stacking(page_height, page_width)
 674    char_polygons: List[Polygon] = []
 675
 676    for box, flattened_text_region in zip(boxes, flattened_text_regions):
 677        assert flattened_text_region.height + flattened_text_regions_double_pad == box.height
 678        assert flattened_text_region.width + flattened_text_regions_double_pad == box.width
 679
 680        up = box.up + flattened_text_regions_pad + page_pad
 681        left = box.left + flattened_text_regions_pad + page_pad
 682
 683        box = Box(
 684            up=up,
 685            down=up + flattened_text_region.height - 1,
 686            left=left,
 687            right=left + flattened_text_region.width - 1,
 688        )
 689
 690        box.fill_image(
 691            image,
 692            flattened_text_region.flattened_image,
 693            image_mask=flattened_text_region.flattened_mask,
 694        )
 695
 696        if flattened_text_region.flattened_char_polygons:
 697            for char_polygon in flattened_text_region.flattened_char_polygons:
 698                char_polygons.append(char_polygon.to_shifted_polygon(
 699                    offset_y=up,
 700                    offset_x=left,
 701                ))
 702
 703    return image, boxes, char_polygons

 714    @classmethod
 715    def generate_precise_text_region_candidate_polygons(
 716        cls,
 717        precise_mask: Mask,
 718        disconnected_text_region_mask: Mask,
 719    ):
 720        assert precise_mask.box and disconnected_text_region_mask.box
 721
 722        # Get the intersection.
 723        intersected_box = Box(
 724            up=max(precise_mask.box.up, disconnected_text_region_mask.box.up),
 725            down=min(precise_mask.box.down, disconnected_text_region_mask.box.down),
 726            left=max(precise_mask.box.left, disconnected_text_region_mask.box.left),
 727            right=min(precise_mask.box.right, disconnected_text_region_mask.box.right),
 728        )
 729        assert intersected_box.up <= intersected_box.down
 730        assert intersected_box.left <= intersected_box.right
 731
 732        precise_mask = intersected_box.extract_mask(precise_mask)
 733        disconnected_text_region_mask = intersected_box.extract_mask(disconnected_text_region_mask)
 734
 735        # Apply mask bitwise-and operation.
 736        intersected_mask = Mask(
 737            mat=(disconnected_text_region_mask.mat & precise_mask.mat).astype(np.uint8)
 738        )
 739        intersected_mask = intersected_mask.to_box_attached(intersected_box)
 740
 741        # NOTE:
 742        # 1. Could extract more than one polygons.
 743        # 2. Some polygons are in border and should be removed later.
 744        return intersected_mask.to_disconnected_polygons()

 746    @classmethod
 747    def strtree_query_intersected_polygons(
 748        cls,
 749        strtree: STRtree,
 750        id_to_anchor_polygon: Dict[int, Polygon],
 751        candidate_polygon: Polygon,
 752    ):
 753        candidate_shapely_polygon = candidate_polygon.to_shapely_polygon()
 754        candidate_mask = candidate_polygon.mask
 755
 756        for anchor_shapely_polygon in strtree.query(candidate_shapely_polygon):
 757            anchor_id = id(anchor_shapely_polygon)
 758            anchor_polygon = id_to_anchor_polygon[anchor_id]
 759            anchor_mask = anchor_polygon.mask
 760
 761            intersected_ratio = calculate_boxed_masks_intersected_ratio(
 762                anchor_mask=anchor_mask,
 763                candidate_mask=candidate_mask,
 764                use_candidate_as_base=True,
 765            )
 766
 767            yield (
 768                anchor_id,
 769                anchor_polygon,
 770                anchor_mask,
 771                candidate_mask,
 772                intersected_ratio,
 773            )

 795    def build_flattened_text_regions(
 796        self,
 797        page_image: Image,
 798        page_text_region_infos: Sequence[PageTextRegionInfo],
 799        page_non_text_region_polygons: Sequence[Polygon],
 800        rng: RandomGenerator,
 801    ):
 802        text_region_polygon_dilate_ratio = float(
 803            rng.uniform(
 804                self.config.text_region_flattener_text_region_polygon_dilate_ratio_min,
 805                self.config.text_region_flattener_text_region_polygon_dilate_ratio_max,
 806            )
 807        )
 808        typical_long_side_ratio_min = \
 809            self.config.text_region_flattener_typical_long_side_ratio_min
 810
 811        text_region_polygons: List[Polygon] = []
 812        grouped_char_polygons: List[Sequence[Polygon]] = []
 813        for page_text_region_info in page_text_region_infos:
 814            text_region_polygons.append(page_text_region_info.precise_text_region_polygon)
 815            grouped_char_polygons.append(page_text_region_info.char_polygons)
 816
 817        # Inject nagative regions.
 818        for page_non_text_region_polygon in page_non_text_region_polygons:
 819            text_region_polygons.append(page_non_text_region_polygon)
 820            grouped_char_polygons.append(tuple())
 821
 822        text_region_flattener = TextRegionFlattener(
 823            typical_long_side_ratio_min=typical_long_side_ratio_min,
 824            text_region_polygon_dilate_ratio=text_region_polygon_dilate_ratio,
 825            image=page_image,
 826            text_region_polygons=text_region_polygons,
 827            grouped_char_polygons=grouped_char_polygons,
 828            is_training=True,
 829        )
 830
 831        # Resize positive ftr.
 832        positive_flattened_text_regions: List[FlattenedTextRegion] = []
 833        # For negative sampling.
 834        positive_reference_heights: List[float] = []
 835        positive_reference_widths: List[float] = []
 836        num_negative_flattened_text_regions = 0
 837
 838        for flattened_text_region in text_region_flattener.flattened_text_regions:
 839            if not flattened_text_region.flattened_char_polygons:
 840                num_negative_flattened_text_regions += 1
 841                continue
 842
 843            char_height_median = flattened_text_region.get_char_height_meidan()
 844
 845            text_region_resize_char_height_median = int(
 846                rng.integers(
 847                    self.config.text_region_resize_char_height_median_min,
 848                    self.config.text_region_resize_char_height_median_max + 1,
 849                )
 850            )
 851            scale = text_region_resize_char_height_median / char_height_median
 852
 853            height, width = flattened_text_region.shape
 854            resized_height = round(height * scale)
 855            resized_width = round(width * scale)
 856
 857            flattened_text_region = flattened_text_region.to_resized_flattened_text_region(
 858                resized_height=resized_height,
 859                resized_width=resized_width,
 860            )
 861
 862            positive_reference_heights.append(resized_height)
 863            positive_reference_widths.append(resized_width)
 864
 865            # Post rotate.
 866            post_rotate_angle = 0
 867            if flattened_text_region.is_typical:
 868                if rng.random() < self.config.text_region_typical_post_rotate_prob:
 869                    # Upside down only.
 870                    post_rotate_angle = 180
 871            else:
 872                if rng.random() < self.config.text_region_untypical_post_rotate_prob:
 873                    # 3-way rotate.
 874                    post_rotate_angle = rng_choice(rng, (180, 90, 270), probs=(0.5, 0.25, 0.25))
 875
 876            if post_rotate_angle != 0:
 877                flattened_text_region = \
 878                    flattened_text_region.to_post_rotated_flattened_text_region(post_rotate_angle)
 879
 880            positive_flattened_text_regions.append(flattened_text_region)
 881
 882        # Resize negative ftr.
 883        negative_reference_heights = list(
 884            rng_choice_with_size(
 885                rng,
 886                positive_reference_heights,
 887                size=num_negative_flattened_text_regions,
 888                replace=(num_negative_flattened_text_regions > len(positive_reference_heights)),
 889            )
 890        )
 891
 892        negative_height_max = max(positive_reference_heights)
 893        negative_width_max = max(positive_reference_widths)
 894
 895        negative_flattened_text_regions: List[FlattenedTextRegion] = []
 896
 897        for flattened_text_region in text_region_flattener.flattened_text_regions:
 898            if flattened_text_region.flattened_char_polygons:
 899                continue
 900
 901            reference_height = negative_reference_heights.pop()
 902            scale = reference_height / flattened_text_region.height
 903
 904            height, width = flattened_text_region.shape
 905            resized_height = round(height * scale)
 906            resized_width = round(width * scale)
 907
 908            # Remove negative region that is too large.
 909            if resized_height > negative_height_max or resized_width > negative_width_max:
 910                continue
 911
 912            flattened_text_region = flattened_text_region.to_resized_flattened_text_region(
 913                resized_height=resized_height,
 914                resized_width=resized_width,
 915            )
 916
 917            # Post rotate.
 918            post_rotate_angle = 0
 919            if flattened_text_region.is_typical:
 920                if rng.random() < self.config.text_region_typical_post_rotate_prob:
 921                    # Upside down only.
 922                    post_rotate_angle = 180
 923            else:
 924                if rng.random() < self.config.text_region_untypical_post_rotate_prob:
 925                    # 3-way rotate.
 926                    post_rotate_angle = rng_choice(rng, (180, 90, 270), probs=(0.5, 0.25, 0.25))
 927
 928            if post_rotate_angle != 0:
 929                flattened_text_region = \
 930                    flattened_text_region.to_post_rotated_flattened_text_region(post_rotate_angle)
 931
 932            negative_flattened_text_regions.append(flattened_text_region)
 933
 934        flattened_text_regions = (
 935            *positive_flattened_text_regions,
 936            *negative_flattened_text_regions,
 937        )
 938        return flattened_text_regions

 940    def run(self, input: PageTextRegionStepInput, rng: RandomGenerator):
 941        page_distortion_step_output = input.page_distortion_step_output
 942        page_image = page_distortion_step_output.page_image
 943        page_char_polygon_collection = page_distortion_step_output.page_char_polygon_collection
 944        page_disconnected_text_region_collection = \
 945            page_distortion_step_output.page_disconnected_text_region_collection
 946        page_non_text_region_collection = \
 947            page_distortion_step_output.page_non_text_region_collection
 948
 949        page_resizing_step_output = input.page_resizing_step_output
 950        page_resized_text_line_mask = page_resizing_step_output.page_text_line_mask
 951
 952        debug = None
 953        if self.config.debug:
 954            debug = DebugPageTextRegionStep()
 955
 956        # Build R-tree to track text regions.
 957        # https://github.com/shapely/shapely/issues/640
 958        id_to_disconnected_text_region_polygon: Dict[int, Polygon] = {}
 959        disconnected_text_region_shapely_polygons: List[ShapelyPolygon] = []
 960
 961        for polygon in page_disconnected_text_region_collection.to_polygons():
 962            shapely_polygon = polygon.to_shapely_polygon()
 963            id_to_disconnected_text_region_polygon[id(shapely_polygon)] = polygon
 964            disconnected_text_region_shapely_polygons.append(shapely_polygon)
 965
 966        disconnected_text_region_tree = STRtree(disconnected_text_region_shapely_polygons)
 967
 968        # Get the precise text regions.
 969        precise_text_region_candidate_polygons: List[Polygon] = []
 970        for resized_precise_polygon in page_resized_text_line_mask.to_disconnected_polygons():
 971            # Resize back to the shape after distortion.
 972            precise_polygon = resized_precise_polygon.to_conducted_resized_polygon(
 973                page_resized_text_line_mask,
 974                resized_height=page_image.height,
 975                resized_width=page_image.width,
 976            )
 977
 978            # Find and extract intersected text region.
 979            # NOTE: One precise_polygon could be overlapped with
 980            # more than one disconnected_text_region_polygon!
 981            for _, _, disconnected_text_region_mask, precise_mask, _ in \
 982                    self.strtree_query_intersected_polygons(
 983                        strtree=disconnected_text_region_tree,
 984                        id_to_anchor_polygon=id_to_disconnected_text_region_polygon,
 985                        candidate_polygon=precise_polygon,
 986                    ):
 987                precise_text_region_candidate_polygons.extend(
 988                    self.generate_precise_text_region_candidate_polygons(
 989                        precise_mask=precise_mask,
 990                        disconnected_text_region_mask=disconnected_text_region_mask,
 991                    )
 992                )
 993
 994        if debug:
 995            debug.page_image = page_image
 996            debug.precise_text_region_candidate_polygons = precise_text_region_candidate_polygons
 997
 998        # Help gc.
 999        del id_to_disconnected_text_region_polygon
1000        del disconnected_text_region_shapely_polygons
1001        del disconnected_text_region_tree
1002
1003        # Bind char-level polygon to precise text region.
1004        id_to_precise_text_region_polygon: Dict[int, Polygon] = {}
1005        precise_text_region_shapely_polygons: List[ShapelyPolygon] = []
1006
1007        for polygon in precise_text_region_candidate_polygons:
1008            shapely_polygon = polygon.to_shapely_polygon()
1009            id_to_precise_text_region_polygon[id(shapely_polygon)] = polygon
1010            precise_text_region_shapely_polygons.append(shapely_polygon)
1011
1012        precise_text_region_tree = STRtree(precise_text_region_shapely_polygons)
1013
1014        id_to_char_polygons: DefaultDict[int, List[Polygon]] = defaultdict(list)
1015        for char_polygon in page_char_polygon_collection.polygons:
1016            best_precise_text_region_id = None
1017            intersected_ratio_max = 0
1018
1019            for (
1020                precise_text_region_id,
1021                _,
1022                _,
1023                _,
1024                intersected_ratio,
1025            ) in self.strtree_query_intersected_polygons(
1026                strtree=precise_text_region_tree,
1027                id_to_anchor_polygon=id_to_precise_text_region_polygon,
1028                candidate_polygon=char_polygon,
1029            ):
1030                if intersected_ratio > intersected_ratio_max:
1031                    intersected_ratio_max = intersected_ratio
1032                    best_precise_text_region_id = precise_text_region_id
1033
1034            if best_precise_text_region_id is not None:
1035                id_to_char_polygons[best_precise_text_region_id].append(char_polygon)
1036            else:
1037                # NOTE: Text line with only a small char (i.e. delimiter) could enter this branch.
1038                # In such case, the text line bounding box is smaller than the char polygon, since
1039                # the leading/trailing char paddings are ignored during text line rendering.
1040                # It's acceptable for now since: 1) this case happens rarely, 2) and it won't
1041                # introduce labeling noise.
1042                logger.warning(f'Cannot assign a text region for char_polygon={char_polygon}')
1043
1044        page_text_region_infos: List[PageTextRegionInfo] = []
1045        for precise_text_region_shapely_polygon in precise_text_region_shapely_polygons:
1046            ptrsp_id = id(precise_text_region_shapely_polygon)
1047            if ptrsp_id not in id_to_char_polygons:
1048                # Not related to any char polygons.
1049                continue
1050            assert id_to_char_polygons[ptrsp_id]
1051            page_text_region_infos.append(
1052                PageTextRegionInfo(
1053                    precise_text_region_polygon=id_to_precise_text_region_polygon[ptrsp_id],
1054                    char_polygons=id_to_char_polygons[ptrsp_id],
1055                )
1056            )
1057
1058        # Help gc.
1059        del id_to_precise_text_region_polygon
1060        del precise_text_region_shapely_polygons
1061        del precise_text_region_tree
1062
1063        if debug:
1064            debug.page_text_region_infos = page_text_region_infos
1065
1066        # Negative sampling.
1067        page_non_text_region_polygons = self.sample_page_non_text_region_polygons(
1068            page_non_text_region_polygons=tuple(page_non_text_region_collection.to_polygons()),
1069            num_page_text_region_infos=len(page_text_region_infos),
1070            rng=rng,
1071        )
1072
1073        flattened_text_regions = self.build_flattened_text_regions(
1074            page_image=page_image,
1075            page_text_region_infos=page_text_region_infos,
1076            page_non_text_region_polygons=page_non_text_region_polygons,
1077            rng=rng,
1078        )
1079        if debug:
1080            debug.flattened_text_regions = flattened_text_regions
1081
1082        # Stack text regions.
1083        image, _, char_polygons = stack_flattened_text_regions(
1084            page_pad=0,
1085            flattened_text_regions_pad=self.config.stack_flattened_text_regions_pad,
1086            flattened_text_regions=flattened_text_regions,
1087        )
1088
1089        # Post uniform rotation.
1090        shape_before_rotate = image.shape
1091        rotate_angle = 0
1092
1093        if self.config.enable_post_rotate:
1094            rotate_angle = int(
1095                rng.integers(
1096                    self.config.post_rotate_angle_min,
1097                    self.config.post_rotate_angle_max + 1,
1098                )
1099            )
1100            rotated_result = rotate.distort(
1101                {'angle': rotate_angle},
1102                image=image,
1103                polygons=char_polygons,
1104            )
1105            assert rotated_result.image and rotated_result.polygons
1106            image = rotated_result.image
1107            char_polygons = rotated_result.polygons
1108
1109        return PageTextRegionStepOutput(
1110            page_image=image,
1111            page_char_polygons=char_polygons,
1112            shape_before_rotate=shape_before_rotate,
1113            rotate_angle=rotate_angle,
1114            debug=debug,
1115        )

256    @classmethod
257    def process_text_region_polygons(
258        cls,

260        shape: Tuple[int, int],
261        text_region_polygons: Sequence[Polygon],
262        force_no_dilation_flags: Optional[Sequence[bool]] = None,
263    ):
264        text_mask = Mask.from_polygons(shape, text_region_polygons)

277            assert len(force_no_dilation_flags) == len(text_region_polygons)
278            force_no_dilation_flags_iter = force_no_dilation_flags

284
285            if not force_no_dilation_flag:
286                # Dilate.
287                text_region_polygon = text_region_polygon.to_dilated_polygon(
288                    ratio=text_region_polygon_dilate_ratio,
289                )

291

294                text_region_polygon.to_bounding_rectangular_polygon(shape)

301            original_text_region_polygon.fill_mask(bounding_other_text_mask, 0)
302
303            # Get protentially dilated text region.
304            bounding_text_mask = Mask.from_shapable(bounding_other_text_mask)
305            bounding_text_mask = bounding_text_mask.to_box_attached(bounding_box)
306            # NOTE: Use the protentially dilated text region polygon to set the current text mask.
307            text_region_polygon.fill_mask(bounding_text_mask, value=1)
308
309            # Should not use the protentially dilated text region polygon anymore.
310            del text_region_polygon
311
312            # Trim protentially dilated text region polygon by eliminating other text region.
313            bounding_trimmed_text_mask = Mask.from_masks(
314                bounding_box,
315                [
316                    # Includes the protentially dilated text region.
317                    bounding_text_mask,
318                    # But not includes any other text regions.
319                    bounding_other_text_mask.to_inverted_mask(),
320                ],
321                ElementSetOperationMode.INTERSECT,
322            )
323
324            # Get non-text region.
325            bounding_non_text_mask = bounding_rectangular_polygon.extract_mask(non_text_mask)
326
327            # Combine trimmed text region and non-text region.
328            bounding_extended_text_region_mask = Mask.from_masks(
329                bounding_box,
330                [bounding_trimmed_text_mask, bounding_non_text_mask],
331            )
332
333            bounding_extended_text_region_masks.append(bounding_extended_text_region_mask)
334            bounding_rectangular_polygons.append(bounding_rectangular_polygon)
335
336        return bounding_extended_text_region_masks, bounding_rectangular_polygons

366            else:
367                # Reference line (p0 -> p3).

731        assert intersected_box.left <= intersected_box.right

914                resized_height=resized_height,

922                    # Upside down only.
923                    post_rotate_angle = 180

926                    # 3-way rotate.

931                    flattened_text_region.to_post_rotated_flattened_text_region(post_rotate_angle)

937            *negative_flattened_text_regions,
