 76    def lazy_post_init(self):
 77        initialized = (self._distance is not None)
 78        if initialized:
 79            return

 94    @classmethod
 95    def calculate_theta_delta(
 96        cls,
 97        vector0: 'Vector',
 98        vector1: 'Vector',
 99        clockwise: bool = False,
100    ):
101        theta_delta = (vector1.theta - vector0.theta + PI) % TWO_PI - PI
102        if clockwise and theta_delta < 0:
103            theta_delta += TWO_PI
104        return theta_delta

148    @classmethod
149    def get_bounding_orientation_idx(cls, down_left: Point, down_right: Point):
150        vector = Vector(
151            y=down_right.smooth_y - down_left.smooth_y,
152            x=down_right.smooth_x - down_left.smooth_x,
153        )
154        #        0
155        #  ┌───────────┐
156        #  │           │
157        # 2│           │3
158        #  │           │
159        #  └───────────┘
160        #        1
161        factor = vector.theta / PI
162        if 1.75 <= factor or factor < 0.25:
163            return 1
164        elif 0.25 <= factor < 0.75:
165            return 2
166        elif 0.75 <= factor < 1.25:
167            return 0
168        elif 1.25 <= factor:
169            return 3
170        else:
171            raise RuntimeError()

173    def lazy_post_init(self):
174        if self._bounding_smooth_up is None:
175            self._bounding_smooth_up = min(point.smooth_y for point in self.corner_points)
176            self._bounding_smooth_down = max(point.smooth_y for point in self.corner_points)
177            self._bounding_smooth_left = min(point.smooth_x for point in self.corner_points)
178            self._bounding_smooth_right = max(point.smooth_x for point in self.corner_points)
179            self._bounding_orientation_idx = self.get_bounding_orientation_idx(
180                down_left=self.down_left,
181                down_right=self.down_right,
182            )
183
184        initialized = (self._up_left_vector is not None)
185        if initialized:
186            return
187
188        self._up_left_vector = Vector(
189            y=self.up_left.smooth_y - self.label_point_smooth_y,
190            x=self.up_left.smooth_x - self.label_point_smooth_x,
191        )
192        self._up_right_vector = Vector(
193            y=self.up_right.smooth_y - self.label_point_smooth_y,
194            x=self.up_right.smooth_x - self.label_point_smooth_x,
195        )
196        self._down_right_vector = Vector(
197            y=self.down_right.smooth_y - self.label_point_smooth_y,
198            x=self.down_right.smooth_x - self.label_point_smooth_x,
199        )
200        self._down_left_vector = Vector(
201            y=self.down_left.smooth_y - self.label_point_smooth_y,
202            x=self.down_left.smooth_x - self.label_point_smooth_x,
203        )
204
205        self._up_left_to_up_right_angle = Vector.calculate_theta_delta(
206            self._up_left_vector,
207            self._up_right_vector,
208            clockwise=True,
209        )
210        self._up_right_to_down_right_angle = Vector.calculate_theta_delta(
211            self._up_right_vector,
212            self._down_right_vector,
213            clockwise=True,
214        )
215        self._down_right_to_down_left_angle = Vector.calculate_theta_delta(
216            self._down_right_vector,
217            self._down_left_vector,
218            clockwise=True,
219        )
220        self._down_left_to_up_left_angle = Vector.calculate_theta_delta(
221            self._down_left_vector,
222            self._up_left_vector,
223            clockwise=True,
224        )
225
226        sum_of_angles = sum([
227            self._up_left_to_up_right_angle,
228            self._up_right_to_down_right_angle,
229            self._down_right_to_down_left_angle,
230            self._down_left_to_up_left_angle,
231        ])
232        # Consider valid if deviate within 4 degrees.
233        self._valid = math.isclose(sum_of_angles, TWO_PI, rel_tol=0.012)
234
235        self._clockwise_angle_distribution = normalize_to_probs([
236            self._up_left_to_up_right_angle,
237            self._up_right_to_down_right_angle,
238            self._down_right_to_down_left_angle,
239            self._down_left_to_up_left_angle,
240        ])

260    def to_shifted_page_char_regression_label(self, offset_y: int, offset_x: int):

262
263        # Shift operation doesn't change the lazy fields.
264        shifted = self.copy(with_non_bounding_related_lazy_fields=True)

448    def generate_page_char_regression_labels(
449        self,
450        shape: Tuple[int, int],
451        page_char_polygons: Sequence[Polygon],
452        rng: RandomGenerator,
453    ):
454        page_height, page_width = shape
455
456        # Build a KD tree to for removing deviate point that is too close to another center point.
457        center_points = PointList()
458        for polygon in page_char_polygons:
459            center_points.append(polygon.get_center_point())
460        kd_tree = KDTree(center_points.to_np_array())
461
462        page_char_regression_labels: List[PageCharRegressionLabel] = []
463
464        for char_idx, (polygon, center_point) in enumerate(zip(page_char_polygons, center_points)):
465            assert polygon.num_points == 4
466            up_left, up_right, down_right, down_left = polygon.points
467
468            # 1. The centroid of char polygon.
469            label = PageCharRegressionLabel(
470                char_idx=char_idx,
471                tag=PageCharRegressionLabelTag.CENTROID,
472                label_point_smooth_y=center_point.smooth_y,
473                label_point_smooth_x=center_point.smooth_x,
474                downsampled_label_point_y=center_point.y,
475                downsampled_label_point_x=center_point.x,
476                up_left=up_left,
477                up_right=up_right,
478                down_right=down_right,
479                down_left=down_left,
480            )
481            # The centroid labeling must be valid.
482            assert label.valid
483            page_char_regression_labels.append(label)
484
485            # 2. The deviate points.
486            bounding_box = polygon.bounding_box
487
488            # Sample points in shfited bounding box space.
489            deviate_points_in_bounding_box = PointList()
490            # Some points are invalid, hence multiply the number of samplings by a factor.
491            # Also not to sample the points lying on the border to increase the chance of valid.
492            for _ in range(
493                self.config.num_deviate_char_regression_labels_candiates_factor
494                * self.config.num_deviate_char_regression_labels
495            ):
496                y = int(rng.integers(1, bounding_box.height - 1))
497                x = int(rng.integers(1, bounding_box.width - 1))
498                deviate_points_in_bounding_box.append(Point.create(y=y, x=x))
499
500            # Then transform to the polygon space.
501            np_src_points = np.asarray(
502                [
503                    (0, 0),
504                    (bounding_box.width - 1, 0),
505                    (bounding_box.width - 1, bounding_box.height - 1),
506                    (0, bounding_box.height - 1),
507                ],
508                dtype=np.float32,
509            )
510            np_dst_points = polygon.internals.np_self_relative_points
511            trans_mat = cv.getPerspectiveTransform(
512                np_src_points,
513                np_dst_points,
514                cv.DECOMP_SVD,
515            )
516
517            deviate_points = PointList()
518            for shifted_deviate_point in affine_points(
519                trans_mat,
520                deviate_points_in_bounding_box.to_point_tuple(),
521            ):
522                y = bounding_box.up + shifted_deviate_point.smooth_y
523                x = bounding_box.left + shifted_deviate_point.smooth_x
524                assert 0 <= y < page_height
525                assert 0 <= x < page_width
526                deviate_points.append(Point.create(y=y, x=x))
527
528            # Remove those are too close to another center point.
529            _, np_kd_nbr_indices = kd_tree.query(deviate_points.to_np_array())
530            preserve_flags: List[bool] = [
531                idx == char_idx for idx in np_kd_nbr_indices[:, 0].tolist()
532            ]
533
534            # Build labels.
535            num_valid_deviate_char_regression_labels = 0
536            for deviate_point, preserve_flag in zip(deviate_points, preserve_flags):
537                if num_valid_deviate_char_regression_labels \
538                        >= self.config.num_deviate_char_regression_labels:
539                    break
540
541                if not preserve_flag:
542                    continue
543
544                label = PageCharRegressionLabel(
545                    char_idx=char_idx,
546                    tag=PageCharRegressionLabelTag.DEVIATE,
547                    label_point_smooth_y=deviate_point.smooth_y,
548                    label_point_smooth_x=deviate_point.smooth_x,
549                    downsampled_label_point_y=deviate_point.y,
550                    downsampled_label_point_x=deviate_point.x,
551                    up_left=up_left,
552                    up_right=up_right,
553                    down_right=down_right,
554                    down_left=down_left,
555                )
556                if label.valid:
557                    page_char_regression_labels.append(label)
558                    num_valid_deviate_char_regression_labels += 1
559
560            if num_valid_deviate_char_regression_labels \
561                    < self.config.num_deviate_char_regression_labels:
562                logger.warning(f'Cannot sample enough deviate labels for char_polygon={polygon}')
563
564        return page_char_regression_labels
